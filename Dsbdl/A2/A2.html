import pandas as pd
import numpy as np
from scipy.stats import zscore

# Load the dataset
file_name = 'dataset.csv'
df = pd.read_csv(file_name)

# Display the dataset preview
print("Dataset Preview:")
print(df.head())

# Step 1: Outlier Detection (Before Removing Outliers)
print("\nOutliers Before Detection:")

for column in df.select_dtypes(include=[np.number]).columns:
    # Calculate Q1, Q3, and IQR
    q1 = df[column].quantile(0.25)
    q3 = df[column].quantile(0.75)
    iqr = q3 - q1
    lower_bound = q1 - 1.5 * iqr
    upper_bound = q3 + 1.5 * iqr

    # Identify outliers
    outliers = df[(df[column] < lower_bound) | (df[column] > upper_bound)]
    outlier_count = len(outliers)

    # Display the results
    print(f"\nColumn: {column}")
    print(f"Number of outliers: {outlier_count}")
    if outlier_count > 0:
        print(outliers)
    else:
        print("No outliers detected.")

# Step 2: IQR Method - Removing Outliers
print("\nUsing the IQR Method to Remove Outliers...")
q1 = df['math score'].quantile(0.25)
q3 = df['math score'].quantile(0.75)
iqr = q3 - q1

# Identify and count outliers
iqr_outliers = (df['math score'] < (q1 - 1.5 * iqr)) | (df['math score'] > (q3 + 1.5 * iqr))
print(f"Number of outliers using IQR: {iqr_outliers.sum()}")

# Remove IQR outliers
df_no_outliers_iqr = df[~iqr_outliers]
print(f"New DataFrame size after IQR removal: {df_no_outliers_iqr.shape}")

# Step 3: Z-Score Method - Removing Outliers
print("\nUsing the Z-Score Method to Remove Outliers...")

# Calculate Z-scores for the 'math score' column
z_scores = zscore(df['math score'])

# Identify outliers using Z-scores
z_outliers = (z_scores > 3) | (z_scores < -3)
print(f"Number of outliers using Z-score: {z_outliers.sum()}")

# Remove Z-score outliers
df_no_outliers_zscore = df[~z_outliers]
print(f"New DataFrame size after Z-score removal: {df_no_outliers_zscore.shape}")

# Step 4: Outlier Detection (After Removing Outliers)
print("\nOutliers After Detection:")

for column in df_no_outliers_zscore.select_dtypes(include=[np.number]).columns:
    # Calculate Q1, Q3, and IQR
    q1 = df_no_outliers_zscore[column].quantile(0.25)
    q3 = df_no_outliers_zscore[column].quantile(0.75)
    iqr = q3 - q1
    lower_bound = q1 - 1.5 * iqr
    upper_bound = q3 + 1.5 * iqr

    # Identify outliers
    outliers = df_no_outliers_zscore[(df_no_outliers_zscore[column] < lower_bound) | (df_no_outliers_zscore[column] > upper_bound)]
    outlier_count = len(outliers)

    # Display the results
    print(f"\nColumn: {column}")
    print(f"Number of outliers: {outlier_count}")
    if outlier_count > 0:
        print(outliers)
    else:
        print("No outliers detected.")
