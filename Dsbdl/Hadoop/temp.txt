# Step 1: Create input directory
hadoop fs -mkdir -p /user/hadoop/input

# Step 2: Upload input file
hadoop fs -put /home//Desktop/Hadoop/input/file.txt /user/hadoop/input/

# Step 3: Compile Java files and create JAR
javac -classpath $(hadoop classpath) -d classes WordCount.java TokenizerMapper.java IntSumReducer.java
jar -cvf wordcount.jar -C classes/ .

# Step 4: Run the Hadoop job
hadoop jar wordcount.jar WordCount /user/hadoop/input/file.txt /user/hadoop/output

# Step 5: Check the output
hadoop fs -ls /user/hadoop/output
hadoop fs -cat /user/hadoop/output/part-r-00000

# Step 6: Cleanup (optional)
hadoop fs -rm -r /user/hadoop/output
